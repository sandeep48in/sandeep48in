// V5 SCRIPT: SUPPORTS UNDO/MARK AS DELETED
// COPY ALL OF THIS CODE INTO YOUR GOOGLE APPS SCRIPT EDITOR

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  const lock = LockService.getScriptLock();
  lock.tryLock(10000);

  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName('Expenses');

    if (!sheet) {
      sheet = ss.insertSheet('Expenses');
      // Added 'status' column
      sheet.appendRow(['id', 'date', 'amount', 'type', 'category', 'description', 'status']);
    }

    let data = {};
    
    // 1. Parse JSON
    try {
      if (e.postData && e.postData.contents) {
        data = JSON.parse(e.postData.contents);
      }
    } catch (err) {}

    // 2. Merge Parameters
    if (e.parameter) {
      Object.assign(data, e.parameter);
    }

    // --- ACTION: CLEAR DATA ---
    if (data.action === 'clear') {
      const lastRow = sheet.getLastRow();
      if (lastRow > 1) {
        sheet.deleteRows(2, lastRow - 1);
      }
      return createJSON({ result: 'success', cleared: true });
    }

    // --- ACTION: UNDO (MARK AS DELETED) ---
    if (data.action === 'undo' && data.id) {
      const range = sheet.getDataRange();
      const values = range.getValues();
      
      // Find row with matching ID (Assuming ID is in Col A / Index 0)
      for (let i = 1; i < values.length; i++) {
        if (values[i][0] == data.id) {
          // Write "DELETED" to Col G (Index 6, which is the 7th column)
          sheet.getRange(i + 1, 7).setValue("DELETED"); 
          return createJSON({ result: 'success', status: 'deleted' });
        }
      }
      return createJSON({ result: 'error', message: 'ID not found' });
    }

    // --- ACTION: ADD TRANSACTION ---
    if (data.amount || data.type || data.category) {
      sheet.appendRow([
        data.id || Math.random().toString(36).substring(7),
        data.date || new Date(),
        data.amount || 0,
        data.type || '',
        data.category || '',
        data.description || '',
        'ACTIVE' // Default status
      ]);
      return createJSON({ result: 'success', saved: true });
    }
    
    // --- GET DATA ---
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0];
    const rawData = rows.slice(1).map(row => {
      let obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });

    return createJSON({ data: rawData });

  } catch (err) {
    return createJSON({ result: 'error', message: err.toString() });
  } finally {
    lock.releaseLock();
  }
}

function createJSON(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
